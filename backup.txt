        <button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast"></button>
      </div>
    </div>
  </div>

  <!-- Bootstrap JS -->
  <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>

  <script>
    /* =============================================
       CREATE TICKET MODAL — JavaScript
    ============================================= */

    let currentStep = 1;

    // Enable asset select when client is chosen
    function handleClientChange() {
      const client = document.getElementById('ctClient');
      const asset  = document.getElementById('ctAsset');
      const hint   = document.getElementById('ctAssetHint');
      if (client.value) {
        asset.disabled = false;
        hint.style.display = 'none';
      } else {
        asset.disabled = true;
        asset.value = '';
        hint.style.display = 'block';
      }
    }

    // Step navigation
    function goToStep(step) {
      if (step > currentStep && !validateStep(currentStep)) return;

      // Update panels
      document.querySelectorAll('.ct-step-panel').forEach(p => p.classList.remove('active'));
      document.getElementById('ctPanel' + step).classList.add('active');

      // Update stepper UI
      for (let i = 1; i <= 3; i++) {
        const stepEl   = document.getElementById('ctStep' + i);
        const circleEl = document.getElementById('ctCircle' + i);
        stepEl.classList.remove('active', 'done');
        if (i < step) {
          stepEl.classList.add('done');
          circleEl.innerHTML = '<i class="bi bi-check-lg" style="font-size:14px;"></i>';
        } else if (i === step) {
          stepEl.classList.add('active');
          circleEl.textContent = i;
        } else {
          circleEl.textContent = i;
        }
      }

      // Update connector lines
      document.getElementById('ctLine1').classList.toggle('done', step > 1);
      document.getElementById('ctLine2').classList.toggle('done', step > 2);

      currentStep = step;

      // If step 3, populate review
      if (step === 3) populateReview();
    }

    // Validate current step fields
    function validateStep(step) {
      let valid = true;

      if (step === 1) {
        const title  = document.getElementById('ctTitle');
        const desc   = document.getElementById('ctDescription');
        const client = document.getElementById('ctClient');

        // Title
        if (!title.value.trim()) {
          showError(title, 'ctTitleError'); valid = false;
        } else { clearError(title, 'ctTitleError'); }

        // Description
        if (!desc.value.trim()) {
          showError(desc, 'ctDescError'); valid = false;
        } else { clearError(desc, 'ctDescError'); }

        // Client
        if (!client.value) {
          showError(client, 'ctClientError'); valid = false;
        } else { clearError(client, 'ctClientError'); }
      }

      return valid;
    }

    function showError(el, errId) {
      el.classList.add('is-invalid');
      const err = document.getElementById(errId);
      if (err) err.style.display = 'block';
    }

    function clearError(el, errId) {
      el.classList.remove('is-invalid');
      const err = document.getElementById(errId);
      if (err) err.style.display = 'none';
    }

    // Clear errors on input
    ['ctTitle', 'ctDescription', 'ctClient'].forEach(id => {
      const el = document.getElementById(id);
      if (el) {
        el.addEventListener('input', () => {
          el.classList.remove('is-invalid');
          const errMap = { ctTitle: 'ctTitleError', ctDescription: 'ctDescError', ctClient: 'ctClientError' };
          const errEl = document.getElementById(errMap[id]);
          if (errEl) errEl.style.display = 'none';
        });
        el.addEventListener('change', () => {
          el.classList.remove('is-invalid');
          const errMap = { ctTitle: 'ctTitleError', ctDescription: 'ctDescError', ctClient: 'ctClientError' };
          const errEl = document.getElementById(errMap[id]);
          if (errEl) errEl.style.display = 'none';
        });
      }
    });

    // Populate review step
    function populateReview() {
      const get = id => document.getElementById(id);
      const val = id => get(id).value;
      const txt = id => get(id).options[get(id).selectedIndex]?.text || '—';

      // Title
      get('rv-title').textContent = val('ctTitle') || '—';

      // Category badge
      const cat = val('ctCategory');
      const catIcons = { Hardware: 'bi-cpu', Software: 'bi-laptop', Network: 'bi-wifi', Security: 'bi-shield-lock', Other: 'bi-three-dots' };
      if (cat) {
        get('rv-category').innerHTML = `<span class="ct-badge-category"><i class="bi ${catIcons[cat] || 'bi-tag'}"></i> ${cat}</span>`;
      } else {
        get('rv-category').textContent = '—';
      }

      // Priority badge
      const pri = val('ctPriority');
      const priClass = { Critical: 'ct-priority-critical', High: 'ct-priority-high', Medium: 'ct-priority-medium', Low: 'ct-priority-low' };
      if (pri) {
        get('rv-priority').innerHTML = `<span class="ct-priority-badge ${priClass[pri] || ''}">${pri}</span>`;
      } else {
        get('rv-priority').textContent = '—';
      }

      // Department
      get('rv-department').textContent = val('ctDepartment') || '—';

      // Client
      get('rv-client').textContent = txt('ctClient') !== 'Client' ? txt('ctClient') : '—';

      // Assign
      get('rv-assign').textContent = val('ctAssignTo') || 'Unassigned';

      // Description
      get('rv-description').textContent = val('ctDescription') || '—';

      // Notes
      const notes = val('ctNotes');
      if (notes.trim()) {
        get('rv-notes-wrap').style.display = '';
        get('rv-notes').textContent = notes;
      } else {
        get('rv-notes-wrap').style.display = 'none';
      }

      // Notice
      const contact = val('ctContact');
      const assign  = val('ctAssignTo');
      const noticeTxt = `Your ticket will be assigned to ${assign || 'an IT engineer'} based on the category and priority.${contact ? ` Notifications will be sent via ${contact}.` : ''}`;
      get('rv-notice-text').textContent = noticeTxt;
    }

    // Submit
    function submitTicket() {
      // Here you'd POST to your backend
      const modal = bootstrap.Modal.getInstance(document.getElementById('createTicketModal'));
      modal.hide();

      // Show success toast
      const toastEl = document.getElementById('ctToast');
      const toast = new bootstrap.Toast(toastEl, { delay: 3500 });
      toast.show();

      // Reset after close
      resetModal();
    }

    // Reset everything
    function resetModal() {
      // Reset step
      goToStep(1);
      currentStep = 1;

      // Clear fields
      ['ctTitle','ctDescription','ctNotes'].forEach(id => {
        const el = document.getElementById(id);
        if (el) el.value = '';
      });
      ['ctClient','ctAsset','ctCategory','ctDepartment','ctRepair','ctPriority','ctAssignTo','ctContact'].forEach(id => {
        const el = document.getElementById(id);
        if (el) el.value = '';
      });

      // Reset asset state
      document.getElementById('ctAsset').disabled = true;
      document.getElementById('ctAssetHint').style.display = 'block';

      // Clear all errors
      document.querySelectorAll('.ct-error').forEach(e => e.style.display = 'none');
      document.querySelectorAll('.ct-form-control, .ct-select').forEach(e => e.classList.remove('is-invalid'));

      // Reset stepper visually
      for (let i = 1; i <= 3; i++) {
        const stepEl   = document.getElementById('ctStep' + i);
        const circleEl = document.getElementById('ctCircle' + i);
        stepEl.classList.remove('active', 'done');
        circleEl.textContent = i;
      }
      document.getElementById('ctStep1').classList.add('active');
      document.getElementById('ctLine1').classList.remove('done');
      document.getElementById('ctLine2').classList.remove('done');
    }

    // Reset modal when dismissed via backdrop/Esc
    document.getElementById('createTicketModal').addEventListener('hidden.bs.modal', () => {
      resetModal();
    });
  </script>
